#include "Ui.h"
#include "Validator.h"
void Ui::run()
{
	while (true)
	{
		try
		{
			int cmd = { 0 };
			cout << "Command:";
			cin >> cmd;
			cin.ignore();
			if (cmd == 0)
			{
				break;
			}
			else if (cmd == 1)
			{
				this->addMenu();
			}
			else if (cmd == 2)
			{
				this->printMenu();
			}
			else if (cmd == 3)
			{
				this->removeMenu();
			}
			else if (cmd == 4)
			{
				this->searchMenu();
			}
			else if (cmd == 5)
			{
				this->sizeMenu();
			}
		}
		catch (Exception& e)
		{
			cout << e.getMessage() << endl;
		}
	}
}

void Ui::addMenu()
{
	string jud;
	cout << "Judet:";
	cin >> jud;
	int nr;
	cout << "Number:";
	cin >> nr;
	string chars;
	cout << "Random chars:";
	cin >> chars;
	Plate a{ jud,nr,chars };
	valid(a);
	it.getSet()->add(a);
	//this->repo->add(a);
}

void Ui::printMenu()
{
	//	cout << "Salut";
	this->it.setIterator();
	while (it.valid())
	{
		Plate a = it.getCurrent();
		cout << "Plate:" << a.getJud() << " " << a.getNumber() << " " << a.getRand() << endl;
		it.next();
	}
}

void Ui::removeMenu()
